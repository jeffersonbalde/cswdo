  constructor() {
    super()
    this.attachShadow({ mode: "open" })
    this.profileImageBlobUrl = null // To store the URL of the selected image
  }

  connectedCallback() {
    this.render()
    this.initializeElements()
    this.initEventListeners()
    this.openModal() // Automatically open the modal when the component is created
  }

  disconnectedCallback() {
    // Clean up when modal is removed
    document.body.style.overflow = ""
    if (this.profileImageBlobUrl) {
      URL.revokeObjectURL(this.profileImageBlobUrl) // Clean up blob URL
    }
  }

  initializeElements() {
    // Initialize DOM elements after render
    this.modal = this.shadowRoot.getElementById("addUserModal")
    this.form = this.shadowRoot.getElementById("addUserForm")
    this.userIdInput = this.shadowRoot.getElementById("addUserId")
    this.userTypeInput = this.shadowRoot.getElementById("addUserType")
    this.departmentInput = this.shadowRoot.getElementById("addUserDepartment")
    this.usernameInput = this.shadowRoot.getElementById("addUsername")
    this.handlerNameInput = this.shadowRoot.getElementById("addHandlerName")
    this.passwordInput = this.shadowRoot.getElementById("addPassword")
    this.errorDiv = this.shadowRoot.getElementById("addUserError")
    this.profileImageInput = this.shadowRoot.getElementById("profileImageInput")
    this.profileImagePreview = this.shadowRoot.getElementById("profileImagePreview")
    this.attachFileButton = this.shadowRoot.getElementById("attachFileBtn")
  }

  initEventListeners() {
    // Close modal events
    this.shadowRoot.getElementById("closeAddModal").addEventListener("click", () => {
      this.closeModal()
    })
    this.shadowRoot.getElementById("cancelAddBtn").addEventListener("click", () => {
      this.closeModal()
    })
    // Close on backdrop click
    this.modal.addEventListener("click", (e) => {
      if (e.target === this.modal) {
        this.closeModal()
      }
    })
    // Form submission
    this.form.addEventListener("submit", (e) => {
      e.preventDefault()
      this.handleSubmit()
    })
    // ESC key to close
    document.addEventListener("keydown", (e) => {
      if (e.key === "Escape" && this.modal.classList.contains("show")) {
        this.closeModal()
      }
    })
    // Attach file button click
    this.attachFileButton.addEventListener("click", () => {
      this.profileImageInput.click()
    })
    // Handle file input change
    this.profileImageInput.addEventListener("change", (event) => {
      const file = event.target.files[0]
      if (file) {
        if (this.profileImageBlobUrl) {
          URL.revokeObjectURL(this.profileImageBlobUrl) // Clean up previous blob URL
        }
        this.profileImageBlobUrl = URL.createObjectURL(file)
        this.profileImagePreview.src = this.profileImageBlobUrl
      }
    })
  }

  openModal() {
    this.modal.classList.add("show")
    this.clearError()
    document.body.style.overflow = "hidden"
  }

  closeModal() {
    this.modal.classList.remove("show")
    this.form.reset()
    this.clearError()
    document.body.style.overflow = ""
    // Reset image preview
    if (this.profileImageBlobUrl) {
      URL.revokeObjectURL(this.profileImageBlobUrl)
      this.profileImageBlobUrl = null
    }
    this.profileImagePreview.src = "/placeholder.svg?height=150&width=150" // Reset to default placeholder
    this.remove() // Remove the modal from DOM
  }

  showError(message) {
    this.errorDiv.textContent = message
    this.errorDiv.classList.add("show")
  }

  clearError() {
    this.errorDiv.textContent = ""
    this.errorDiv.classList.remove("show")
  }

  handleSubmit() {
    const formData = new FormData(this.form)
    const data = Object.fromEntries(formData.entries())

    // Add profile image file to data if selected
    if (this.profileImageInput.files && this.profileImageInput.files[0]) {
      data.profileImage = this.profileImageInput.files[0].name // Or the file object itself if needed
    }

    // Basic validation
    if (!data.userId || !data.userType || !data.department || !data.username || !data.handlerName || !data.password) {
      this.showError("Please fill in all required fields.")
      return
    }

    // Process the form data
    console.log("User Data:", data)
    // Dispatch custom event to notify parent component
    this.dispatchEvent(
      new CustomEvent("user-saved", {
        detail: { user: data },
        bubbles: true,
      }),
    )
    alert("User saved successfully!")
    this.closeModal()
  }
